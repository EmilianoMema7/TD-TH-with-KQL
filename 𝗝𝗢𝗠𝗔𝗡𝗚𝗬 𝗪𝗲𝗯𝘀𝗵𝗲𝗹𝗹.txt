let Lookback=30d;
let WebProcesses=dynamic(["apache2","httpd","php-fpm","nginx"]);
let SuspiciousChildren=dynamic(["curl","wget","crontab","useradd","adduser","passwd","iptables","firewall-cmd"]);
let KnownIPs=dynamic(["45.95.147.178","160.119.69.4"]);
union (
DeviceProcessEvents
| where Timestamp>=ago(Lookback)
| where InitiatingProcessFileName in~ (WebProcesses)
| where FileName in~ (SuspiciousChildren)
| where not(ProcessCommandLine has "localhost:42699")
| where not(ProcessCommandLine has "com.instana.plugin.generic.event")
| extend DetectionSource="Process",SuspicionReason="Web server process spawned persistence/download/account/firewall utility"
| project Timestamp,DeviceName,AccountName,DetectionSource,SuspicionReason,FileName,FolderPath,ProcessCommandLine,InitiatingProcessFileName,InitiatingProcessCommandLine,RemoteIP="",RemoteUrl="",SHA256
),
(
DeviceNetworkEvents
| where Timestamp>=ago(Lookback)
| where InitiatingProcessFileName in~ (WebProcesses)
| where RemoteIP in (KnownIPs)
| extend DetectionSource="Network",SuspicionReason="Web server communicated with known JOMANGY infrastructure"
| project Timestamp,DeviceName,AccountName=InitiatingProcessAccountName,DetectionSource,SuspicionReason,FileName=InitiatingProcessFileName,FolderPath=InitiatingProcessFolderPath,ProcessCommandLine=InitiatingProcessCommandLine,InitiatingProcessFileName,InitiatingProcessCommandLine,RemoteIP,RemoteUrl,SHA256=InitiatingProcessSHA256
)
| sort by Timestamp desc
