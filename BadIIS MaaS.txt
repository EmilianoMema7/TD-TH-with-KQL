// Title:
// BadIIS MaaS / Suspicious IIS Worker Process Activity - Threat Hunt
//
// Description:
// Hunt for suspicious Microsoft IIS activity potentially related to BadIIS-style IIS malware,
// including w3wp.exe spawning suspicious child processes, IIS worker outbound traffic to known infrastructure,
// and suspicious IIS file/config modifications.
//
// Initiating Article:
// https://blog.talosintelligence.com/from-pdb-strings-to-maas-tracking-a-commodity-badiis-ecosystem/
//
// MITRE ATT&CK:
// T1505.004 - Server Software Component: IIS Components
// T1505.003 - Server Software Component: Web Shell
// T1059.001 - PowerShell
// T1059.003 - Windows Command Shell
// T1105 - Ingress Tool Transfer
// T1071.001 - Application Layer Protocol: Web Protocols
// T1218 - System Binary Proxy Execution
//
// Expected Result:
// This query may return 0 results in a healthy environment.
// Any result should be reviewed because w3wp.exe rarely spawns scripting engines,
// LOLBins, or communicates with known malicious infrastructure.

let Lookback = 30d;
let BadIISDomains = dynamic([
    "lee.6686ty.vip",
    "iis.01nmwe.xyz"
]);
let BadIISIPs = dynamic([
    "143.92.36.109",
    "38.181.52.147",
    "154.23.186.99",
    "45.194.17.133",
    "154.36.149.4"
]);
let SuspiciousChildren = dynamic([
    "powershell.exe",
    "cmd.exe",
    "rundll32.exe",
    "regsvr32.exe",
    "mshta.exe",
    "certutil.exe",
    "curl.exe",
    "bitsadmin.exe",
    "wscript.exe",
    "cscript.exe"
]);
let SuspiciousTerms = dynamic([
    "downloadstring",
    "invoke-webrequest",
    "invoke-restmethod",
    "frombase64string",
    "encodedcommand",
    "webclient",
    "certutil",
    "bitsadmin",
    "http://",
    "https://"
]);
union
(
    DeviceProcessEvents
    | where Timestamp >= ago(Lookback)
    | where InitiatingProcessFileName =~ "w3wp.exe"
    | where FileName in~ (SuspiciousChildren)
    | where
        ProcessCommandLine has_any (SuspiciousTerms)
        or FileName in~ (
            "rundll32.exe",
            "regsvr32.exe",
            "mshta.exe",
            "certutil.exe",
            "curl.exe",
            "bitsadmin.exe",
            "wscript.exe",
            "cscript.exe"
        )
    | extend DetectionSource = "Process",
             SuspicionReason = "w3wp.exe spawned suspicious child process"
    | project Timestamp,
              DeviceName,
              AccountName,
              DetectionSource,
              SuspicionReason,
              FileName,
              FolderPath,
              ProcessCommandLine,
              InitiatingProcessFileName,
              InitiatingProcessCommandLine,
              RemoteUrl = "",
              RemoteIP = "",
              SHA256
),
(
    DeviceNetworkEvents
    | where Timestamp >= ago(Lookback)
    | where InitiatingProcessFileName =~ "w3wp.exe"
    | where
        RemoteIP in (BadIISIPs)
        or RemoteUrl has_any (BadIISDomains)
    | extend DetectionSource = "Network",
             SuspicionReason = "w3wp.exe connected to known BadIIS-related infrastructure"
    | project Timestamp,
              DeviceName,
              AccountName = InitiatingProcessAccountName,
              DetectionSource,
              SuspicionReason,
              FileName = InitiatingProcessFileName,
              FolderPath = InitiatingProcessFolderPath,
              ProcessCommandLine = InitiatingProcessCommandLine,
              InitiatingProcessFileName,
              InitiatingProcessCommandLine,
              RemoteUrl,
              RemoteIP,
              SHA256 = InitiatingProcessSHA256
)
| sort by Timestamp desc
