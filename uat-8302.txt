(
  DeviceFileEvents
  | where tolower(SHA256) in (UAT8302_SHA256)
  | extend IOCType="SHA256", IOC=tolower(SHA256), Confidence="High"
  | project Timestamp, DeviceName, ActionType, IOCType, IOC, Confidence,
            FileName, FolderPath, InitiatingProcessFileName,
            InitiatingProcessCommandLine, ReportId

(
  DeviceProcessEvents
  | where tolower(SHA256) in (UAT8302_SHA256)
  | extend IOCType="Process SHA256", IOC=tolower(SHA256), Confidence="High"
  | project Timestamp, DeviceName, ActionType, IOCType, IOC, Confidence,
            FileName, FolderPath, InitiatingProcessFileName,
            InitiatingProcessCommandLine=ProcessCommandLine, ReportId

(
  DeviceNetworkEvents
  | extend RemoteUrlLower = tolower(RemoteUrl)
  | where RemoteIP in (UAT8302_IPs)
      or RemoteUrlLower in~ (UAT8302_FullUrls)

      RemoteIP in (UAT8302_IPs), "IP",
      RemoteUrlLower in~ (UAT8302_FullUrls), "Full URL",
